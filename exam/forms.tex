\begin{exercise}\points{10}
  Let $\omega =  3 dxdy + (x^2 -y^3 + z^4) dydz + (x^7-y^5+x) dx dz$.

  Compute $d \omega$.
  \begin{prompt}
  \[
    d \omega = \answer{(2x+5y^4)dx dy dz}
  \]
  \end{prompt}
\end{exercise}

\vfill
