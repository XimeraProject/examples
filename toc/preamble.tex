% Chapter style
\renewcommand{\chapterstyle}{%
\addtocontents{chapter}{1}
    \let\section\section
    \let\subsection\subsection
    \chapter{\thechapter~\@title}
}

% Section style
\renewcommand{\sectionstyle}{
    \titleformat{\title}{\Large\bfseries}{}{0pt}{}
    \renewcommand{\maketitle}{
        \begingroup
        \centering
        \Large\bfseries\@title\par
        \endgroup
    }
    \setcounter{tocdepth}{1}
    \phantomsection\addcontentsline{toc}{section}{\thesection~\@title}%
    \let\section\subsection
    \let\subsection\subsubsection
}
